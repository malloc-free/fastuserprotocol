177 tb_set_addrinfo(tb_session_t *session, int type)
178 {
179  if(session->addr_info == NULL && type > 0)

181  session->addr_hints = malloc(sizeof(struct addrinfo));

184  session->addr_hints->ai_socktype = type;

190  return -1;
191  }
192 
193  int rc = getaddrinfo(session->address, session->port,
194  session->addr_hints, &session->addr_info);
195 
196  if(rc != 0)
197  {
198  PRT_E_S("Error: tb_set_addrinfo: \n", gai_strerror(rc));
199  return -1;
200  }
201 
202  session->status = SESSION_ADDR_RES;
203 
204  return 0;
